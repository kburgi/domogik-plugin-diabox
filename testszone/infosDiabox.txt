function unitChange() {
	var myTempSelect = document.getElementById("temp").value;
	var myWindSelect = document.getElementById("wind").value;
	var myDbxId = getQuerystring("id");
	var myTemplate = getQuerystring("template", null);
	var url = '?id='+myDbxId+'&unitTemp='+myTempSelect+'&unitWind='+myWindSelect;
	if ((myTemplate != null) && (myTemplate != "")) {
		url = url+'&template='+myTemplate;
	}
	window.location.replace(url);

}
